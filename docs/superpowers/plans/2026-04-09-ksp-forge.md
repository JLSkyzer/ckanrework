# KSP Forge Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a modern Electron + React mod manager for KSP1 that uses CKAN metadata and SpaceDock enrichment with a space-themed UI.

**Architecture:** Electron app with main process services (MetaSync, SpaceDock, Resolver, Installer, Profile) communicating via IPC with a React renderer. SQLite for local mod index + cache. CKAN-meta repo cloned locally for mod registry, SpaceDock API for rich content.

**Tech Stack:** Electron, React, TypeScript, Tailwind CSS, Zustand, SQLite (better-sqlite3), Vite (electron-vite), pnpm, Vitest, Playwright

Create `electron/services/database.ts`:

```typescript
import Database from 'better-sqlite3'
import type { ModRow, ModVersionRow, SpaceDockCacheRow, ProfileRow, InstalledModRow } from '../types'

export class DatabaseService {
  private db: Database.Database

  constructor(dbPath: string) {
    this.db = new Database(dbPath)
    this.db.pragma('journal_mode = WAL')
    this.db.pragma('foreign_keys = ON')
    this.init()
  }

  private init() {
    this.db.exec(`
      CREATE TABLE IF NOT EXISTS mods (
        identifier TEXT PRIMARY KEY,
        name TEXT NOT NULL,
        abstract TEXT,
        author TEXT,
        license TEXT,
        latest_version TEXT NOT NULL,
        ksp_version TEXT,
        ksp_version_min TEXT,
        ksp_version_max TEXT,
        download_url TEXT,
        download_size INTEGER,
        spacedock_id INTEGER,
        tags TEXT,
        resources TEXT,
        updated_at INTEGER NOT NULL
      );

      CREATE TABLE IF NOT EXISTS mod_versions (
        identifier TEXT NOT NULL,
        version TEXT NOT NULL,
        ksp_version TEXT,
        ksp_version_min TEXT,
        ksp_version_max TEXT,
        download_url TEXT NOT NULL,
        download_hash TEXT,
        download_size INTEGER,
        depends TEXT,
        recommends TEXT,
        suggests TEXT,
        conflicts TEXT,
        install_directives TEXT NOT NULL,
        PRIMARY KEY (identifier, version),
        FOREIGN KEY (identifier) REFERENCES mods(identifier)
      );

      CREATE TABLE IF NOT EXISTS spacedock_cache (
        spacedock_id INTEGER PRIMARY KEY,
        mod_identifier TEXT NOT NULL,
        description TEXT,
        description_html TEXT,
        background_url TEXT,
        downloads INTEGER,
        followers INTEGER,
        fetched_at INTEGER NOT NULL,
        FOREIGN KEY (mod_identifier) REFERENCES mods(identifier)
      );

      CREATE TABLE IF NOT EXISTS installed_mods (
        profile_id TEXT NOT NULL,
        identifier TEXT NOT NULL,
        version TEXT NOT NULL,
        installed_files TEXT NOT NULL,
        installed_at INTEGER NOT NULL,
        PRIMARY KEY (profile_id, identifier),
        FOREIGN KEY (identifier) REFERENCES mods(identifier)
      );

      CREATE TABLE IF NOT EXISTS profiles (
        id TEXT PRIMARY KEY,
        name TEXT NOT NULL,
        ksp_path TEXT NOT NULL,
        ksp_version TEXT NOT NULL,
        created_at INTEGER NOT NULL,
        updated_at INTEGER NOT NULL
      );

      CREATE VIRTUAL TABLE IF NOT EXISTS mods_fts USING fts5(
        identifier, name, abstract, author,
        content='mods', content_rowid='rowid'
      );

      CREATE TRIGGER IF NOT EXISTS mods_ai AFTER INSERT ON mods BEGIN
        INSERT INTO mods_fts(rowid, identifier, name, abstract, author)
        VALUES (new.rowid, new.identifier, new.name, new.abstract, new.author);
      END;

      CREATE TRIGGER IF NOT EXISTS mods_ad AFTER DELETE ON mods BEGIN
        INSERT INTO mods_fts(mods_fts, rowid, identifier, name, abstract, author)
        VALUES ('delete', old.rowid, old.identifier, old.name, old.abstract, old.author);
      END;

      CREATE TRIGGER IF NOT EXISTS mods_au AFTER UPDATE ON mods BEGIN
        INSERT INTO mods_fts(mods_fts, rowid, identifier, name, abstract, author)
        VALUES ('delete', old.rowid, old.identifier, old.name, old.abstract, old.author);
        INSERT INTO mods_fts(rowid, identifier, name, abstract, author)
        VALUES (new.rowid, new.identifier, new.name, new.abstract, new.author);
      END;
    `)
  }

  listTables(): string[] {
    const rows = this.db.prepare(
      "SELECT name FROM sqlite_master WHERE type='table' AND name NOT LIKE 'sqlite_%' AND name NOT LIKE '%fts%' AND name NOT LIKE '%config%' AND name NOT LIKE '%data%' AND name NOT LIKE '%idx%' AND name NOT LIKE '%content%' AND name NOT LIKE '%docsize%'"
    ).all() as { name: string }[]
    return rows.map(r => r.name)
  }

  upsertMod(mod: ModRow) {
    this.db.prepare(`
      INSERT INTO mods (identifier, name, abstract, author, license, latest_version,
        ksp_version, ksp_version_min, ksp_version_max, download_url, download_size,
        spacedock_id, tags, resources, updated_at)
      VALUES (@identifier, @name, @abstract, @author, @license, @latest_version,
        @ksp_version, @ksp_version_min, @ksp_version_max, @download_url, @download_size,
        @spacedock_id, @tags, @resources, @updated_at)
      ON CONFLICT(identifier) DO UPDATE SET
        name=@name, abstract=@abstract, author=@author, license=@license,
        latest_version=@latest_version, ksp_version=@ksp_version,
        ksp_version_min=@ksp_version_min, ksp_version_max=@ksp_version_max,
        download_url=@download_url, download_size=@download_size,
        spacedock_id=@spacedock_id, tags=@tags, resources=@resources, updated_at=@updated_at
    `).run(mod)
  }

  getMod(identifier: string): ModRow | null {
    return this.db.prepare('SELECT * FROM mods WHERE identifier = ?').get(identifier) as ModRow | null
  }

  getAllMods(limit = 100, offset = 0): ModRow[] {
    return this.db.prepare('SELECT * FROM mods ORDER BY name LIMIT ? OFFSET ?').all(limit, offset) as ModRow[]
  }

  searchMods(query: string): ModRow[] {
    return this.db.prepare(`
      SELECT mods.* FROM mods_fts
      JOIN mods ON mods.identifier = mods_fts.identifier
      WHERE mods_fts MATCH ?
      ORDER BY rank
      LIMIT 100
    `).all(query + '*') as ModRow[]
  }

  upsertModVersion(version: ModVersionRow) {
    this.db.prepare(`
      INSERT INTO mod_versions (identifier, version, ksp_version, ksp_version_min,
        ksp_version_max, download_url, download_hash, download_size, depends,
        recommends, suggests, conflicts, install_directives)
      VALUES (@identifier, @version, @ksp_version, @ksp_version_min, @ksp_version_max,
        @download_url, @download_hash, @download_size, @depends, @recommends,
        @suggests, @conflicts, @install_directives)
      ON CONFLICT(identifier, version) DO UPDATE SET
        ksp_version=@ksp_version, ksp_version_min=@ksp_version_min,
        ksp_version_max=@ksp_version_max, download_url=@download_url,
        download_hash=@download_hash, download_size=@download_size,
        depends=@depends, recommends=@recommends, suggests=@suggests,
        conflicts=@conflicts, install_directives=@install_directives
    `).run(version)
  }

  getModVersions(identifier: string): ModVersionRow[] {
    return this.db.prepare('SELECT * FROM mod_versions WHERE identifier = ? ORDER BY version DESC').all(identifier) as ModVersionRow[]
  }

  upsertSpaceDockCache(cache: SpaceDockCacheRow) {
    this.db.prepare(`
      INSERT INTO spacedock_cache (spacedock_id, mod_identifier, description,
        description_html, background_url, downloads, followers, fetched_at)
      VALUES (@spacedock_id, @mod_identifier, @description, @description_html,
        @background_url, @downloads, @followers, @fetched_at)
      ON CONFLICT(spacedock_id) DO UPDATE SET
        description=@description, description_html=@description_html,
        background_url=@background_url, downloads=@downloads,
        followers=@followers, fetched_at=@fetched_at
    `).run(cache)
  }

  getSpaceDockCache(modIdentifier: string): SpaceDockCacheRow | null {
    return this.db.prepare('SELECT * FROM spacedock_cache WHERE mod_identifier = ?').get(modIdentifier) as SpaceDockCacheRow | null
  }

  getStaleSpaceDockEntries(maxAgeMs: number): string[] {
    const cutoff = Date.now() - maxAgeMs
    const rows = this.db.prepare(`
      SELECT m.identifier FROM mods m
      LEFT JOIN spacedock_cache sc ON m.identifier = sc.mod_identifier
      WHERE m.spacedock_id IS NOT NULL
      AND (sc.fetched_at IS NULL OR sc.fetched_at < ?)
    `).all(cutoff) as { identifier: string }[]
    return rows.map(r => r.identifier)
  }

  createProfile(profile: ProfileRow) {
    this.db.prepare(`
      INSERT INTO profiles (id, name, ksp_path, ksp_version, created_at, updated_at)
      VALUES (@id, @name, @ksp_path, @ksp_version, @created_at, @updated_at)
    `).run(profile)
  }

  getProfiles(): ProfileRow[] {
    return this.db.prepare('SELECT * FROM profiles ORDER BY name').all() as ProfileRow[]
  }

  getProfile(id: string): ProfileRow | null {
    return this.db.prepare('SELECT * FROM profiles WHERE id = ?').get(id) as ProfileRow | null
  }

  deleteProfile(id: string) {
    this.db.prepare('DELETE FROM installed_mods WHERE profile_id = ?').run(id)
    this.db.prepare('DELETE FROM profiles WHERE id = ?').run(id)
  }

  addInstalledMod(mod: InstalledModRow) {
    this.db.prepare(`
      INSERT INTO installed_mods (profile_id, identifier, version, installed_files, installed_at)
      VALUES (@profile_id, @identifier, @version, @installed_files, @installed_at)
      ON CONFLICT(profile_id, identifier) DO UPDATE SET
        version=@version, installed_files=@installed_files, installed_at=@installed_at
    `).run(mod)
  }

  removeInstalledMod(profileId: string, identifier: string) {
    this.db.prepare('DELETE FROM installed_mods WHERE profile_id = ? AND identifier = ?').run(profileId, identifier)
  }

  getInstalledMods(profileId: string): InstalledModRow[] {
    return this.db.prepare('SELECT * FROM installed_mods WHERE profile_id = ?').all(profileId) as InstalledModRow[]
  }

  getModCount(): number {
    return (this.db.prepare('SELECT COUNT(*) as count FROM mods').get() as { count: number }).count
  }

  close() {
    this.db.close()
  }
}
```

```typescript
  }

export class MetaSyncService {
  private git: SimpleGit
  private repoPath: string
  private db: DatabaseService

  constructor(repoPath: string, db: DatabaseService) {
    this.repoPath = repoPath
    this.db = db
    this.git = simpleGit()
  }

  async sync(onProgress?: (current: number, total: number) => void): Promise<number> {
    if (!fs.existsSync(path.join(this.repoPath, '.git'))) {
      await this.git.clone(CKAN_META_REPO, this.repoPath, ['--depth', '1'])
    } else {
      this.git.cwd(this.repoPath)
      await this.git.pull()
    }

    return this.indexAll(onProgress)
  }

  private indexAll(onProgress?: (current: number, total: number) => void): number {
    const entries = fs.readdirSync(this.repoPath, { withFileTypes: true })
    const modDirs = entries.filter(e => e.isDirectory() && !e.name.startsWith('.'))
    let indexed = 0

    for (const dir of modDirs) {
      const modPath = path.join(this.repoPath, dir.name)
      const ckanFiles = fs.readdirSync(modPath).filter(f => f.endsWith('.ckan'))

      for (const file of ckanFiles) {
        try {
          const content = fs.readFileSync(path.join(modPath, file), 'utf-8')
          const ckan = JSON.parse(content) as CkanMetadata
          const { mod, version } = parseCkanFile(ckan)
          this.db.upsertMod(mod)
          this.db.upsertModVersion(version)
        } catch {
          // Skip malformed files
        }
      }

      indexed++
      onProgress?.(indexed, modDirs.length)
    }

    return indexed
  }
}
```

```typescript
import type { SpaceDockCacheRow } from '../types'
import type { DatabaseService } from './database'

const SPACEDOCK_API = 'https://spacedock.info/api'
const CACHE_TTL_MS = 24 * 60 * 60 * 1000 // 24 hours

interface SpaceDockModResponse {
  name: string
  description: string
  description_html: string
  background: string | null
  downloads: number
  followers: number
  versions: Array<{
    friendly_version: string
    game_version: string
    download_path: string
    changelog: string
    created: string
  }>
}

export class SpaceDockService {
  private db: DatabaseService

  constructor(db: DatabaseService) {
    this.db = db
  }

  async fetchModData(modIdentifier: string): Promise<SpaceDockCacheRow | null> {
    // Check cache first
    const cached = this.db.getSpaceDockCache(modIdentifier)
    if (cached && Date.now() - cached.fetched_at < CACHE_TTL_MS) {
      return cached
    }

    // Get spacedock_id from mod
    const mod = this.db.getMod(modIdentifier)
    if (!mod?.spacedock_id) return null

    try {
      const response = await fetch(`${SPACEDOCK_API}/mod/${mod.spacedock_id}`)
      if (!response.ok) return cached ?? null

      const data: SpaceDockModResponse = await response.json()

      const cacheRow: SpaceDockCacheRow = {
        spacedock_id: mod.spacedock_id,
        mod_identifier: modIdentifier,
        description: data.description,
        description_html: data.description_html,
        background_url: data.background,
        downloads: data.downloads,
        followers: data.followers,
        fetched_at: Date.now()
      }

      this.db.upsertSpaceDockCache(cacheRow)
      return cacheRow
    } catch {
      return cached ?? null
    }
  }

  async fetchBatch(identifiers: string[]): Promise<void> {
    for (const id of identifiers) {
      await this.fetchModData(id)
      // Rate limit: 100ms between requests
      await new Promise(resolve => setTimeout(resolve, 100))
    }
  }
}
```

```typescript
import type { ModVersionRow, Relationship } from '../types'
import type { DatabaseService } from './database'

export interface ResolvedMod {
  identifier: string
  version: string
  download_url: string
  download_size: number | null
  download_hash: string | null
  install_directives: string
  isDependency: boolean
}

export interface ResolutionResult {
  success: boolean
  toInstall: ResolvedMod[]
  conflicts: string[]
  missing: string[]
  warnings: string[]
}

export class ResolverService {
  private db: DatabaseService

  constructor(db: DatabaseService) {
    this.db = db
  }

  resolve(identifiers: string[], kspVersion: string, profileId?: string): ResolutionResult {
    const toInstall = new Map<string, ResolvedMod>()
    const conflicts: string[] = []
    const missing: string[] = []
    const warnings: string[] = []
    const visited = new Set<string>()

    // Get already installed mods
    const installed = new Set<string>()
    if (profileId) {
      for (const mod of this.db.getInstalledMods(profileId)) {
        installed.add(mod.identifier)
      }
    }

    // Resolve each requested mod
    for (const id of identifiers) {
      this.resolveOne(id, kspVersion, toInstall, conflicts, missing, warnings, visited, installed, false)
    }

    // Check for conflicts between all mods to install
    const allMods = [...toInstall.values()]
    for (const mod of allMods) {
      const versions = this.db.getModVersions(mod.identifier)
      const ver = versions.find(v => v.version === mod.version)
      if (!ver?.conflicts) continue

      const modConflicts: Relationship[] = JSON.parse(ver.conflicts)
      for (const conflict of modConflicts) {
        if (toInstall.has(conflict.name) || installed.has(conflict.name)) {
          conflicts.push(`${mod.identifier} conflicts with ${conflict.name}`)
        }
      }
    }

    return {
      success: conflicts.length === 0 && missing.length === 0,
      toInstall: allMods,
      conflicts,
      missing,
      warnings
    }
  }

  private resolveOne(
    identifier: string,
    kspVersion: string,
    toInstall: Map<string, ResolvedMod>,
    conflicts: string[],
    missing: string[],
    warnings: string[],
    visited: Set<string>,
    installed: Set<string>,
    isDependency: boolean
  ) {
    if (visited.has(identifier)) return
    visited.add(identifier)

    if (installed.has(identifier)) return

    const versions = this.db.getModVersions(identifier)
    if (versions.length === 0) {
      missing.push(identifier)
      return
    }

    // Pick best compatible version
    const compatible = this.findCompatibleVersion(versions, kspVersion)
    const selected = compatible ?? versions[0]

    if (!compatible) {
      warnings.push(`${identifier} v${selected.version} may not be compatible with KSP ${kspVersion}`)
    }

    toInstall.set(identifier, {
      identifier,
      version: selected.version,
      download_url: selected.download_url,
      download_size: selected.download_size,
      download_hash: selected.download_hash,
      install_directives: selected.install_directives,
      isDependency
    })

    // Resolve dependencies
    if (selected.depends) {
      const deps: Relationship[] = JSON.parse(selected.depends)
      for (const dep of deps) {
        this.resolveOne(dep.name, kspVersion, toInstall, conflicts, missing, warnings, visited, installed, true)
      }
    }
  }

  private findCompatibleVersion(versions: ModVersionRow[], kspVersion: string): ModVersionRow | null {
    for (const v of versions) {
      if (this.isKspCompatible(v, kspVersion)) return v
    }
    return null
  }

  private isKspCompatible(version: ModVersionRow, kspVersion: string): boolean {
    if (version.ksp_version && version.ksp_version === kspVersion) return true
    if (version.ksp_version === 'any') return true

    const target = this.parseVersion(kspVersion)
    if (!target) return true // Can't parse, assume compatible

    if (version.ksp_version_min) {
      const min = this.parseVersion(version.ksp_version_min)
      if (min && this.compareVersions(target, min) < 0) return false
    }

    if (version.ksp_version_max) {
      const max = this.parseVersion(version.ksp_version_max)
      if (max && this.compareVersions(target, max) > 0) return false
    }

    if (!version.ksp_version && !version.ksp_version_min && !version.ksp_version_max) {
      return true // No version constraint
    }

    if (version.ksp_version && version.ksp_version !== kspVersion) {
      // Check major.minor match
      const mod = this.parseVersion(version.ksp_version)
      if (mod && target && mod[0] === target[0] && mod[1] === target[1]) return true
      return false
    }

    return true
  }

  private parseVersion(v: string): number[] | null {
    const parts = v.split('.').map(Number)
    if (parts.some(isNaN)) return null
    return parts
  }

  private compareVersions(a: number[], b: number[]): number {
    const len = Math.max(a.length, b.length)
    for (let i = 0; i < len; i++) {
      const av = a[i] ?? 0
      const bv = b[i] ?? 0
      if (av !== bv) return av - bv
    }
    return 0
  }
}
```

```typescript
export class InstallerService {
  private db: DatabaseService

  constructor(db: DatabaseService) {
    this.db = db
  }

  buildInstallPlan(mods: ResolvedMod[]): InstallPlanItem[] {
    return mods.map(mod => ({
      identifier: mod.identifier,
      version: mod.version,
      downloadUrl: mod.download_url,
      totalSize: mod.download_size,
      hash: mod.download_hash,
      directives: this.parseDirectives(mod.install_directives),
      isDependency: mod.isDependency
    }))
  }

  parseDirectives(json: string): InstallDirective[] {
    return JSON.parse(json) as InstallDirective[]
  }

  async installMod(
    item: InstallPlanItem,
    kspPath: string,
    profileId: string,
    tempDir: string,
    onProgress?: (downloaded: number, total: number | null) => void
  ): Promise<string[]> {
    // 1. Download
    const zipPath = path.join(tempDir, `${item.identifier}-${item.version}.zip`)
    await this.download(item.downloadUrl, zipPath, item.totalSize, onProgress)

    // 2. Verify hash
    if (item.hash) {
      const fileHash = await this.computeHash(zipPath)
      if (fileHash !== item.hash) {
        fs.unlinkSync(zipPath)
        throw new Error(`Hash mismatch for ${item.identifier}: expected ${item.hash}, got ${fileHash}`)
      }
    }

    // 3. Extract
    const extractDir = path.join(tempDir, `${item.identifier}-extract`)
    await this.extractZip(zipPath, extractDir)

    // 4. Install files per directives
    const installedFiles: string[] = []
    for (const directive of item.directives) {
      const files = await this.applyDirective(directive, extractDir, kspPath)
      installedFiles.push(...files)
    }

    // 5. Track in DB
    const record: InstalledModRow = {
      profile_id: profileId,
      identifier: item.identifier,
      version: item.version,
      installed_files: JSON.stringify(installedFiles),
      installed_at: Date.now()
    }
    this.db.addInstalledMod(record)

    // 6. Cleanup
    fs.rmSync(extractDir, { recursive: true, force: true })
    fs.unlinkSync(zipPath)

    return installedFiles
  }

  async uninstallMod(profileId: string, identifier: string, kspPath: string): Promise<void> {
    const installed = this.db.getInstalledMods(profileId)
    const mod = installed.find(m => m.identifier === identifier)
    if (!mod) return

    const files: string[] = JSON.parse(mod.installed_files)

    // Remove files (reverse order to handle nested dirs)
    for (const file of files.reverse()) {
      const fullPath = path.join(kspPath, file)
      if (fs.existsSync(fullPath)) {
        const stat = fs.statSync(fullPath)
        if (stat.isFile()) {
          fs.unlinkSync(fullPath)
        }
      }
    }

    // Clean up empty directories
    for (const file of files) {
      const dir = path.dirname(path.join(kspPath, file))
      if (fs.existsSync(dir) && fs.readdirSync(dir).length === 0) {
        fs.rmSync(dir, { recursive: true })
      }
    }

    this.db.removeInstalledMod(profileId, identifier)
  }

  private async download(
    url: string,
    destPath: string,
    totalSize: number | null,
    onProgress?: (downloaded: number, total: number | null) => void
  ): Promise<void> {
    const response = await fetch(url, { redirect: 'follow' })
    if (!response.ok) throw new Error(`Download failed: ${response.status} ${response.statusText}`)

    const reader = response.body!.getReader()
    const writer = createWriteStream(destPath)
    let downloaded = 0

    while (true) {
      const { done, value } = await reader.read()
      if (done) break
      writer.write(Buffer.from(value))
      downloaded += value.byteLength
      onProgress?.(downloaded, totalSize)
    }

    writer.end()
    await new Promise<void>((resolve, reject) => {
      writer.on('finish', resolve)
      writer.on('error', reject)
    })
  }

  private async computeHash(filePath: string): Promise<string> {
    const hash = createHash('sha256')
    const stream = createReadStream(filePath)
    for await (const chunk of stream) {
      hash.update(chunk)
    }
    return hash.digest('hex').toUpperCase()
  }

  private async extractZip(zipPath: string, destDir: string): Promise<void> {
    fs.mkdirSync(destDir, { recursive: true })
    return new Promise((resolve, reject) => {
      createReadStream(zipPath)
        .pipe(Extract({ path: destDir }))
        .on('close', resolve)
        .on('error', reject)
    })
  }

  private async applyDirective(
    directive: InstallDirective,
    extractDir: string,
    kspPath: string
  ): Promise<string[]> {
    const targetBase = path.join(kspPath, directive.install_to)
    const installedFiles: string[] = []

    // Find the source directory/file in extracted contents
    let sourcePath: string | null = null

    if (directive.find) {
      sourcePath = this.findInDir(extractDir, directive.find)
    } else if (directive.file) {
      const candidate = path.join(extractDir, directive.file)
      if (fs.existsSync(candidate)) sourcePath = candidate
    }

    if (!sourcePath) return installedFiles

    // Copy to target
    const destPath = path.join(targetBase, path.basename(sourcePath))
    this.copyRecursive(sourcePath, destPath, directive, installedFiles, kspPath)

    return installedFiles
  }

  private findInDir(dir: string, name: string): string | null {
    const entries = fs.readdirSync(dir, { withFileTypes: true })
    for (const entry of entries) {
      if (entry.name === name) return path.join(dir, entry.name)
      if (entry.isDirectory()) {
        const found = this.findInDir(path.join(dir, entry.name), name)
        if (found) return found
      }
    }
    return null
  }

  private copyRecursive(
    src: string,
    dest: string,
    directive: InstallDirective,
    installedFiles: string[],
    kspPath: string
  ) {
    const stat = fs.statSync(src)

    if (stat.isFile()) {
      // Check filters
      if (directive.filter?.includes(path.basename(src))) return
      fs.mkdirSync(path.dirname(dest), { recursive: true })
      fs.copyFileSync(src, dest)
      installedFiles.push(path.relative(kspPath, dest))
    } else if (stat.isDirectory()) {
      fs.mkdirSync(dest, { recursive: true })
      installedFiles.push(path.relative(kspPath, dest))
      for (const entry of fs.readdirSync(src)) {
        this.copyRecursive(
          path.join(src, entry),
          path.join(dest, entry),
          directive,
          installedFiles,
          kspPath
        )
      }
    }
  }
}
```

```typescript
      id: randomUUID(),
      name: newName,
      ksp_path: source.ksp_path,
      ksp_version: source.ksp_version,
      created_at: Date.now(),
      updated_at: Date.now()

    const mods = this.db.getInstalledMods(profileId)
    return {
      profile,
      mods: mods.map(m => ({ identifier: m.identifier, version: m.version }))
    }
  }

  detectKspVersion(kspPath: string): string {
    // Try readme.txt first
    const readmePath = path.join(kspPath, 'readme.txt')
    if (fs.existsSync(readmePath)) {
      const content = fs.readFileSync(readmePath, 'utf-8')
      const match = content.match(/Version\s+([\d.]+)/)
      if (match) return match[1]
    }

    // Try buildID.txt
    const buildPath = path.join(kspPath, 'buildID.txt')
    if (fs.existsSync(buildPath)) {
      const content = fs.readFileSync(buildPath, 'utf-8')
      const match = content.match(/build id\s*=\s*\d+\s*[\r\n]+.*?([\d.]+)/i)
      if (match) return match[1]
    }

    return 'unknown'
  }

  validateKspPath(kspPath: string): boolean {
    return fs.existsSync(path.join(kspPath, 'GameData'))
  }
```

```tsx
import { useUiStore } from '../../stores/ui-store'
import type { ModVersionRow, Relationship } from '../../../electron/types'

export function ModDependencies({ versions }: { versions: ModVersionRow[] }) {
  const { openModDetail } = useUiStore()
  const latest = versions[0]
  if (!latest) return <p className="text-[#64748b]">No version data available.</p>

  const recommends: Relationship[] = latest.recommends ? JSON.parse(latest.recommends) : []
  const suggests: Relationship[] = latest.suggests ? JSON.parse(latest.suggests) : []
  const conflicts: Relationship[] = latest.conflicts ? JSON.parse(latest.conflicts) : []

  function renderList(title: string, items: Relationship[], color: string) {
    if (items.length === 0) return null
    return (
      <div className="mb-4">
        <h3 className={`text-sm font-semibold mb-2`} style={{ color }}>{title}</h3>
        <div className="space-y-1">

              {item.min_version && <span className="text-[#64748b] ml-2">≥ {item.min_version}</span>}
              {item.max_version && <span className="text-[#64748b] ml-2">≤ {item.max_version}</span>}
            </button>
          ))}
        </div>
      </div>
    )
  }

  return (
    <div>
      {depends.length === 0 && recommends.length === 0 && suggests.length === 0 && conflicts.length === 0 ? (
        <p className="text-[#64748b]">This mod has no dependencies.</p>
      ) : (

          {renderList('Required', depends, '#e2e8f0')}

          {renderList('Suggested', suggests, '#94a3b8')}
          {renderList('Conflicts', conflicts, '#ef4444')}
        </>
      )}
    </div>
  )
}
```

```tsx
import { useState } from 'react'
import { api } from '../lib/ipc'
import { useProfileStore } from '../stores/profile-store'
import type { ResolutionResult } from '../../electron/services/resolver'

export function useInstall() {
  const [resolution, setResolution] = useState<ResolutionResult | null>(null)
  const [showDialog, setShowDialog] = useState(false)
  const [installing, setInstalling] = useState(false)
  const [progress, setProgress] = useState<{ current: number; total: number }>({ current: 0, total: 0 })
  const { activeProfileId, fetchInstalledMods } = useProfileStore()

  async function requestInstall(identifiers: string[]) {
    if (!activeProfileId) return
    const profile = await api.profiles.get(activeProfileId)
    if (!profile) return

    const result = await api.resolver.resolve(identifiers, profile.ksp_version, activeProfileId)
    setResolution(result)
    setShowDialog(true)
  }

  async function confirmInstall() {
    if (!resolution || !activeProfileId) return
    const profile = await api.profiles.get(activeProfileId)
    if (!profile) return

    setInstalling(true)
    setProgress({ current: 0, total: resolution.toInstall.length })

    try {
      const tempDir = await api.profiles.get(activeProfileId).then(() => {
        // Use app temp directory
        return `${profile.ksp_path}/../ksp-forge-temp`
      })

      await api.installer.install(
        JSON.stringify(resolution.toInstall),
        profile.ksp_path,
        activeProfileId,
        tempDir
      )

      await fetchInstalledMods()
    } finally {
      setInstalling(false)
      setShowDialog(false)
      setResolution(null)
    }
  }

  function cancelInstall() {
    setShowDialog(false)
    setResolution(null)
  }

  return { resolution, showDialog, installing, progress, requestInstall, confirmInstall, cancelInstall }
}
```

```tsx
import { Component, type ReactNode, type ErrorInfo } from 'react'

interface Props { children: ReactNode }
interface State { hasError: boolean; error: Error | null }

export class ErrorBoundary extends Component<Props, State> {
  state: State = { hasError: false, error: null }

  static getDerivedStateFromError(error: Error) {
    return { hasError: true, error }
  }

  componentDidCatch(error: Error, info: ErrorInfo) {
    console.error('UI Error:', error, info)
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="h-screen bg-[#0d0d1a] flex items-center justify-center">
          <div className="text-center max-w-md">
            <h1 className="text-xl font-bold text-[#ef4444] mb-3">Something went wrong</h1>
            <p className="text-sm text-[#94a3b8] mb-4">{this.state.error?.message}</p>
            <button
              onClick={() => this.setState({ hasError: false, error: null })}
              className="px-4 py-2 bg-[#6366f1] text-white rounded-lg text-sm"
            >
              Try Again
            </button>
          </div>
        </div>
      )
    }
    return this.props.children
  }
}
```
